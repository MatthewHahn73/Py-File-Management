        <ul>

            <li>Installation: <a href="https://www.python.org/downloads/">Link</a></li>
        </ul>

        <ul>
            <li>PYQT6</li> 
            <ul>
                <li>Purpose: GUI</li>
                <li>Installation: <a href="https://pypi.org/project/pyqt6/">Link</a></li>
            </ul>
            <li>Paramiko</li> 
            <ul>
                <li>Purpose: SSH Connections</li>

            </ul>    
        </ul>
    </ul>
